dane$ownHome <- gsub('own', "", as.character(dane$ownHome))

Zależność pomiędzy przychodem, a cen

H0 - przychód wpływa na cenę H1 - przychód nie wpływa na ceną

shapiro.test(dane$income)
## 
##  Shapiro-Wilk normality test
## 
## data:  dane$income
## W = 0.9712, p-value = 0.00001031
# Zmienna 'income' nie należy do rozkładu normalnego. 
# Z tego powodu należy wykonać test Kruskala-Wallisa.
kruskal.test(dane$income ~ dane$segment)
## 
##  Kruskal-Wallis rank sum test
## 
## data:  dane$income by dane$segment
## Kruskal-Wallis chi-squared = 128.09, df = 3, p-value <
## 0.00000000000000022
# p-value jest na bardzo niskim poziomie, co oznacza, 
# że średnie są od siebie niezależne.
pairwise.wilcox.test(dane$income, dane$segment, p.adjust.method = "BH")
